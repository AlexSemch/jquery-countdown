jQuery Countdown

General Setup

Add the following lines to your <head>

<script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.6.4/jquery.min.js"></script>
<script type="text/javascript" src="js/jquery.countdown.js"></script>

4:3:2:1 (4 days, 3 hours, 2 minutes, 1 second)

$(document).ready(function() {
    $('#counter1').countdown({
        startTime: "4:3:2:1",
        stepTime: 1,
        image: "images/digits.png"
    });
});

April 22 (from Date object)

$(document).ready(function() {
    var _date = new Date();
    _date.setMonth(3);
    _date.setDate(22);
    _date.setHours(0);
    _date.setMinutes(0);
    _date.setSeconds(0);
    $('#counter2').countdown({
        startTime: _date,
        stepTime: 1,
        image: "images/digits.png"
    });
});
